Euclidian Algorithm


                #include

                    int gcd(int a, int b){
                        if(a == 0)
                        return b;
                        else
                        return (a,b%a);
                    }
                    
                    int main(){
                        int a,b,g;
                        printf("Enter first number:\n");
                        scanf("%d",&a);
                        printf("Enter second numver:\n");
                        scanf("%d",&b);
                        g = gcd(a,b);
                        printf("The gcd of %d and %d = %d\n",a,b,g);
                        return 0;
                    }
