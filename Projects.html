Fuzzy Set

    #include<stdio.h>
    float min(float a, float b){
        if(a < b)
            return a;
        else
            return b;
    }
    
    float max(float a, float b){
        if(a > b)
            return a;
        else
            return b;
    }
    
    int main(){
        float x, y, fh, fl, fuz;
        float z = 1.0;
        printf("Enter membership function of the first set:\n");
        scanf("%f", &x);
        printf("Enter membership function of the second set:\n");
        scanf("%f", &y);
    
        fl = min(x, y);
        fh = max(x, y);
    
        fuz = z - x;
    
        printf("The function of intersection = %f\n", fl);
        printf("The function of union = %f\n", fh);
        printf("The function of complement = %f\n", fuz);
        return 0;
    }

Euclidian Algorithm

      #include

          int gcd(int a, int b){
              if(a == 0)
              return b;
              else
              return (a,b%a);
          }
          
          int main(){
              int a,b,g;
              printf("Enter first number:\n");
              scanf("%d",&a);
              printf("Enter second numver:\n");
              scanf("%d",&b);
              g = gcd(a,b);
              printf("The gcd of %d and %d = %d\n",a,b,g);
              return 0;
          }

IMPLICATION Operation

      #include
          int Imp(int a, int b){
              if(a==1 && b == 0)
              return 0;
              else
              return 1;
          }
          
          int main(){
              int i,j;
              printf("A\tB\tA-->B\n\n");
              for(i=0;i<=1; i++){
                  for(j=0;j<=1; j++){
                      printf("%d\t%d\t%d\n\n",i,j,Imp(i,j));
                  }
              }
              return 0;
          }
